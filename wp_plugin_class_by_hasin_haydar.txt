functions.php তে লিখতে হবে

function before_add_p_tag(){
    echo "<p>";
}
add_action('before_action', 'before_add_p_tag');

function after_add_p_tag(){
    echo "</p>";
}
add_action('after_content', 'after_add_p_tag');

result: <p>some content</p>
